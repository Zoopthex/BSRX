add_namespace = phoney


#Event for Italy after Ethiopian war
#activate SLOG and kick Italy from faction with Germany

country_event = {
	id = phoney.1
	title = phoney.1.t
	desc = phoney.1.d

	picture = GFX_report_war_in_ethiopia_is_a_slog

	fire_only_once = yes

	trigger = {
		tag = ITA
		OR = {
			AND = {
				ITA = {
					has_full_control_of_state = 271
					has_full_control_of_state = 1111
					has_full_control_of_state = 1113
				}
			}
			AND = {
				ETH = {
					has_full_control_of_state = 271
					has_full_control_of_state = 1111
					has_full_control_of_state = 1113
					is_puppet_of = ITA
				}
			}
		}
		has_war = no #War in Ethiopia is over
		date < 1938.12.30 #failsafe for future scenarios #Italian East Africa supposed to exist in 1939
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		ITA = {
			annex_country = {
				target = ETH
				transfer_troops = no
			}
		}
	}

	option = {
		name = phoney.1.a
		ai_chance = { 
			factor = 75
			modifier = {
				factor = 100
				is_historical_focus_on = yes
			}
		}	
		leave_faction = yes
		add_ideas = ITA_italy_unprep
		set_global_flag = italy_unprepared_for_war
	}

	option = {
		name = phoney.1.b
		ai_chance = { 
			factor = 25
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		add_stability = -0.1
	}
}

#Italy is now prepared for war - deactivate SLOG event
country_event = {
	id = phoney.2
	title = phoney.2.t
	desc = phoney.2.d

	picture = GFX_report_italy_is_ready_for_war

	fire_only_once = yes

	trigger = {
		tag = ITA
		is_historical_focus_on = yes
		has_global_flag = italy_unprepared_for_war
		GER = { controls_state = 16 }
		OR = {
			date > 1940.6.1
			has_war = yes #it's war anyway so get rid of it
			NOT = { has_government = fascism } #communist/democratic Italy must get rid of this without triggering the war
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = phoney.2.a
		leave_faction = yes
		remove_ideas = ITA_italy_unprep
		clr_global_flag = italy_unprepared_for_war
		if = {
			limit = { 
				ITA = { has_government = fascism } 
				NOT = { 
					is_in_faction_with = ENG
					is_in_faction_with = FRA
				}
			}
			country_event = {
				id = phoney.3
				days = 2
			}
		}
	}
}
