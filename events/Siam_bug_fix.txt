#Siam actually attacks Allies when in faction with Japan

add_namespace = supersiam

country_event = {
	id = supersiam.1
	title = supersiam.1.t
	desc = supersiam.1.d
	picture = GFX_report_asia_1936
	
	fire_only_once = yes

	trigger = {
		tag = SIA
		has_war = no
		is_in_faction_with = JAP
		has_government = fascism
		JAP = { has_war_with = ENG }
		JAP = { is_ai = yes } #Player can prevent Siam from joining the war
		is_historical_focus_on = yes
	}
	
	option = {
		name = supersiam.1.a
		ai_chance = { factor = 100 }
		add_to_war = { 
			targeted_alliance = JAP 
			enemy = ENG
			hostility_reason = asked_to_join
		}
	}

	option = {
		name = supersiam.1.b
	}	
}

	id = supersiam.16
	title = supersiam.16.t
	desc = supersiam.16.d
	picture = GFX_report_asia_1936
	
	fire_only_once = yes

	trigger = {
		tag = JAP
		date > 1939.1.1
		date < 1943.1.1
		has_government = fascism
		has_global_flag = phibun_rules #Phibun must be a ruler of Siam
		NOT = { is_in_faction_with = SIA } #If player has already joined it won't trigger
		OR = {
			has_war_with = ENG
			SIA = { has_war_with = ENG }
		}
	}

	option = {
		name = supersiam.16.a
		ai_chance = { 
			factor = 50
			modifier = {
				factor = 100
				is_historical_focus_on = yes
			}
		}
		SIA = { 
			country_event = { 
				id = supersiam.17 
				hours = 6 
			}
		}
	}

	option = {
		name = supersiam.16.b
		ai_chance = { 
			factor = 50
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}		
	}
}
