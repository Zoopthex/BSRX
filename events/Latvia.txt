# Ulmanis Dictatorship or Free Elections?
country_event = {
	id = latvia.5
	title = latvia.5.t
	desc = latvia.5.d
	picture = GFX_report_event_latvia_ulmanis_handshake
	
	trigger = {
		tag = LAT
		has_idea = LAT_Ulmanis_Coup
		date > 1936.03.19
	}
	
	fire_only_once = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = latvia.5.a
		ai_chance = { factor = 6 }
		remove_ideas = LAT_Ulmanis_Coup
		if = {
			limit = { NOT = { has_government = neutrality } }
			set_politics = {
				ruling_party = neutrality
				elections_allowed = no
			}
		}
		create_country_leader = {
			name = "Karlis Ulmanis"
			desc = "POLITICS_KARLIS_ULMANIS_DESC"
			picture = "GFX_Portrait_latvia_karlis_ulmanis"
			expire = "1965.1.1"
			ideology = despotism
			traits = {
				LAT_Rural_Militarist
			}
		}
		set_country_flag = LAT_Constitutional_Norms_Ignored
	}
	
	option = {
		name = latvia.5.b
		ai_chance = {
			factor = 5
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		remove_ideas = LAT_Ulmanis_Coup
		remove_ideas = LAT_Economic_Recovery
		set_politics = {
			ruling_party = democratic
			elections_allowed = yes
		}
		effect_tooltip = {
			create_country_leader = {
				name = "Pauls Kalniņš"
				desc = ""
				picture = "GFX_Portrait_latvia_Pauls_Kalnins"
				expire = "1965.1.1"
				ideology = socialism
				traits = {
					great_compromiser
				}
			}
		}
		set_country_flag = LAT_New_Elections_Held
	}
}

#Baltic Entente to join faction together
country_event = {
	id = latvia.9
	title = latvia.9.t
	desc = latvia.9.d
	picture = GFX_report_event_generic_read_write
	
	is_triggered_only = yes
	
	option = {
		name = latvia.9.a
		ai_chance = { 
			factor = 95
			modifier = {
				factor = 1.35
				LAT = { has_idea_with_trait = POL_smooth_talking_charmer }
			}
		}
		if = {
			limit = {
				EST = {
					has_idea = LAT_Baltic_Entente_Member
					is_in_faction_with = LAT
					NOT = {
						any_other_country = {
							has_war_with = EST
							NOT = { has_war_with = ROOT }
						}
					}
				}
			}
			hidden_effect = { LAT = { remove_from_faction = EST } }
			add_to_faction = EST
			add_ai_strategy = {
				type = alliance
				id = "EST"
				value = 175
			}
		}
		if = {
			limit = {
				LIT = {
					has_idea = LAT_Baltic_Entente_Member
					is_in_faction_with = LAT
					NOT = {
						any_other_country = {
							has_war_with = LIT
							NOT = { has_war_with = ROOT }
						}
					}
				}
			}
			hidden_effect = { LAT = { remove_from_faction = LIT } }
			add_to_faction = LIT
			add_ai_strategy = {
				type = alliance
				id = "LIT"
				value = 175
			}
		}
		LAT = {
			dismantle_faction = yes
			country_event = latvia.7
		}
		add_to_faction = LAT
		add_ai_strategy = {
			type = alliance
			id = "LAT"
			value = 175
		}
	}

	option = {
		name = latvia.9.b
		ai_chance = { factor = 5 }
		LAT = { country_event = latvia.8 }
	}
}
