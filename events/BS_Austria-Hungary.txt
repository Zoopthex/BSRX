#The Question of Transylvania
country_event = {
	id = AUH.3
	title = AUH.3.t
	desc = AUH.3.d
	picture = GFX_report_event_hungary_parliament
	
	is_triggered_only = yes

	option = { #Slow but easy way
		trigger = {
			NOT = { country_exists = TRA }
		}
		name = AUH.3.a

		TRA = {
			transfer_state = 76
			transfer_state = 82
			transfer_state = 83
			transfer_state = 84
			transfer_state = 1201
			transfer_state = 1507
		}
		add_state_core = 1201
		add_state_core = 1625
		add_state_core = 1643
		puppet = TRA
		TRA = { add_ideas = HUN_crownland_integration }
		hidden_effect = {
			set_country_flag = HUN_autonomous_TRA
		}
		if = {
			limit = {
				NOT = { has_dlc = "Together for Victory" }
			}
			hidden_effect = {
				TRA = {
					set_country_flag = {
						flag = HUN_integration_time
						days = 180
						value = 1
					}
				}
			}
		}
		ai_chance = {
			factor = 75
		}
	}

	option = { #With compliance, but hard
		trigger = {
			has_dlc = "La Resistance"
		}
		name = AUH.3.b

		add_state_core = 1201
		add_state_core = 1625
		add_state_core = 1643
		hidden_effect = {
			set_country_flag = HUN_TRA_compliance
		}
		ai_chance = {
			factor = 25
		}
	}
}
