    title = bs_australia.1.t

    trigger = {
		tag = AST

				traits = {
					Treasurer
				}
			}
			country_event = { id = bs_australia.2 days = 19 }
			set_party_name = { ideology = democratic long_name = AST_democratic_party_2_long name = AST_democratic_party_2 }
		}
	}

    option = {
        name = bs_australia.1.a

    }
}
